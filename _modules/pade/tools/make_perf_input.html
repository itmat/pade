[docs]def read_input(filename):
    with open(filename) as f:
        f.next()    
        rows = []

        for line in f:
            line = line.rstrip()
            row = line.split("\t")
            row = row[1:] + row[1:9]
            rows.append("\t".join(row))
    
        return rows
